@Service("riverService")
public class RiverServiceImpl implements RiverService.Iface {
    private static final Logger replayLogger = LoggerFactory.getLogger("logger_wmrec_river_replaylog");
    
    @Override
    public Response recall(Request request) throws TException {
        ContextBase contextBase = initContextBase(request);// 初始化 ContextBase
        // ... 业务逻辑、图策略处理逻辑、日志写入逻辑
        
        Response response = contextBase.getRiverResponse();
        
        raptorResponse(response, contextBase);// 埋点
        raptorDuration(contextBase, start);// 埋点
        return response;
    }
    
    @Override
    private ContextBase initContextBase(Request request) {
        ContextBase contextBase = new ContextBase();
        contextBase.setRequest(request);
        
        Response response = new Response();
        contextBase.setResponse(response);
        
        contextBase.setExtraParams(MapUtils.emptyIfNull(request.getExtraParams()));
        
        return contextBase;
    }
    
    private void raptorResponse(Response response, ContextBase contextBase) {
        boolean responseEmpty = false;
        if (response == null || response.getPayloadsSize() == 0) {
            responseEmpty = true;
        } else {
            for (ByteBuffer payload : response.getPayloads()) {
                if (payload == null) {
                    responseEmpty = true;
                    break;
                }
            }
        }
        Cat.logEvent(CatKeyConst.SERVICE_RESPONSE_EMPTY, contextBase.getGraphScene(), responseEmpty ? "1" : "0", "");
    }
    
    private void raptorDuration(ContextBase contextBase, long start) {
        if (CatKeyConst.RIVER_PHF_APPKEY.equals(ContextBase.APPKEY)) {
            String sceneTag = PhfCommonUtil.getSceneTag(contextBase.getGraphScene(), contextBase.getPageSource());
            Cat.newCompletedTransactionWithDuration("biz.phf.scene.time", sceneTag, System.currentTimeMillis() - start);
        }
        if (!CatKeyConst.RIVER_SQS_APPKEY.equals(ContextBase.APPKEY)
                || contextBase.getExperimentResponse() == null
                || CollectionUtils.isEmpty(contextBase.getExperimentResponse().getExperimentInfoList())) {
            return;
        }

        try {
            ExperimentInfo experimentInfo = contextBase.getExperimentResponse().getExperimentInfoList().get(0);
            Cat.newCompletedTransactionWithDuration(CatKeyConst.GRAPH_COST_BY_EXPERIMENT,
                    contextBase.getGraphScene() + "." + experimentInfo.getGroupId() + "_" + experimentInfo.getBucketId(),
                    System.currentTimeMillis() - start);
        } catch (Exception e) {
            LOG.warn("raptorDuration failed");
        }
    }
}
